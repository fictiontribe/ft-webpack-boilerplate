#Usage \n

# 1. Clone the boilerplate: \n

git clone https://github.com/fictiontribe/ft-webpack-boilerplate.git

# 2. Install dependencies: \n

npm install

# 3. Start a development server: \n

npm run start

# 4. Create a production build: \n

npm run build

# 5. Deploy build: \n

npm run deploy
